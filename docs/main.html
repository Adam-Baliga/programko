 48class camera:
 49    """ 
 50    class to store the position and rotation of the camera
 51
 52    To change the position and rotation of the camera use the translate and rotate methods
 53
 54    Attributes:
 55        position: numpy array of shape (3,) representing the position of the camera
 56        rotation: numpy array of shape (3,3) representing the rotation matrix
 57
 58    Example:
 59
 60    cam = camera(np.array([0,0,-5]))
 61    cam.rotate(np.array([[-1,0,0],[0,1,0],[0,0,1]])))
 62
 63    """
 64    def __init__(self,position=np.array([0,0,0])):
 65        self.position = position
 66        self.rotation = np.eye(3)
 67    def rotate(self,rotation_matrix):
 68        """
 69        Rotate the camera by the given rotation matrix
 70
 71        rotation_matrix: numpy array of shape (3,3)
 72        """
 73        self.rotation = rotation_matrix @ self.rotation
 74    def translate(self,translation_vector):
 75        """
 76        Translate the camera by the given translation vector
 77        translation_vector: numpy array of shape (3,)
 78        """
 79        self.position = self.position + translation_vector

256        """
257        Opens a dialog to get the object name, angle and rotation plane and rotates the object by that angle in that plane
258        positive angle -> counter clockwise
259        negative angle -> clockwise
260        """
261
262        object_name,angle,rotation_plane = get_rotation(objects=self.objects)
263
264        angle = np.radians(angle)
265
266        if object_name in self.objects:
267            rot_mat = rotation.rotation_matrix(angle,rotation_plane,inverse=False)
268            inverse_rot_mat = rotation.rotation_matrix(angle,rotation_plane,inverse=True)
269
270
271            self.objects[object_name].rotate(rot_mat,inverse_rot_mat)
272            self.log.write(f"Rotated object {object_name} by {np.degrees(angle)} degrees in the {rotation_plane} plane")
273
274        else:
275            print(f"Object {object_name} not found")

277    def combine_objects(self,operation):
278        """
279        Opens a dialog to get the names of the two objects to be combined and combines them using the given operation
280
281        operation: string, can be "union","difference","intersection"
282        """
283        result = get_2_objects(title=operation,objects=self.objects)
284        
285        if result == None:
286            return
287        
288        obj_name1,obj_name2 = result[0],result[1]
289
290        if obj_name1 in self.objects and obj_name2 in self.objects:
291            if operation == "union":
292                self.objects[f"combined-union{self.object_id}"] = csg.CSG_union(self.objects[obj_name1],self.objects[obj_name2])
293            elif operation == "difference":
294                self.objects[f"combined-difference{self.object_id}"] = csg.CSG_difference(self.objects[obj_name1],self.objects[obj_name2])
295            elif operation == "intersection":
296                self.objects[f"combined-intersection{self.object_id}"] = csg.CSG_intersection(self.objects[obj_name1],self.objects[obj_name2])
297
298            self.log.write(f"Combined objects {obj_name1} and {obj_name2} using {operation} to create combined-{operation}{self.object_id}")
299            self.object_id += 1
300
301            #remove the original objects
302            del self.objects[obj_name1]
303            del self.objects[obj_name2]
304        else:
305            print(f"One or both objects not found: {obj_name1}, {obj_name2}")

257        """

273            self.log.write(f"Rotated object {object_name} by {np.degrees(angle)} degrees in the {rotation_plane} plane")

275        else:
276            print(f"Object {object_name} not found")

278    def combine_objects(self,operation):
279        """
280        Opens a dialog to get the names of the two objects to be combined and combines them using the given operation
281
282        operation: string, can be "union","difference","intersection"
283        """
284        result = get_2_objects(title=operation,objects=self.objects)
285        
286        if result == None:
287            return
288        
289        obj_name1,obj_name2 = result[0],result[1]
290
291        if obj_name1 in self.objects and obj_name2 in self.objects:
292            if operation == "union":
293                self.objects[f"combined-union{self.object_id}"] = csg.CSG_union(self.objects[obj_name1],self.objects[obj_name2])
294            elif operation == "difference":
295                self.objects[f"combined-difference{self.object_id}"] = csg.CSG_difference(self.objects[obj_name1],self.objects[obj_name2])
296            elif operation == "intersection":
297                self.objects[f"combined-intersection{self.object_id}"] = csg.CSG_intersection(self.objects[obj_name1],self.objects[obj_name2])
298
299            self.log.write(f"Combined objects {obj_name1} and {obj_name2} using {operation} to create combined-{operation}{self.object_id}")
300            self.object_id += 1
301
302            #remove the original objects
303            del self.objects[obj_name1]
304            del self.objects[obj_name2]
305        else:
306            print(f"One or both objects not found: {obj_name1}, {obj_name2}")
